# 輸入中文名字英文名字跟年齡
```
import java.util.Scanner;
public class ex000{
	public static void main(String[] args){
		Scanner scn = new Scanner(System.in);
		System.out.print("輸入您的名字:");
		String name = scn.nextLine();
		System.out.print("輸入您的英文名字:");
		String ename = scn.nextLine();
		System.out.print("輸入你的年齡: ");
		int num = scn.nextInt();
		System.out.printf("你好， %s %n你的英文名字: %s %n你的年齡: %d 歲", name, ename, num);
		scn.close();
	}
}
```

# 比大小  ==> 取最大 或 取最小
# 2個數字取最大
```
import java.util.Scanner;
public class ex003 {
	public static void main(String[] args) {
		Scanner scn = new Scanner(System.in);
		int num1, num2, max;
		System.out.print("請輸入2個數字(用空白鍵區隔) : ");
		num1 = scn.nextInt();
		num2 = scn.nextInt();
		
		if (num1 > num2){
			max = num1;
		}else{
			max = num2;
		}
		System.out.println("最大的數字是 : "+ max);
		scn.close();
	}
}
```

# 3個數字同時取最大與最小
```
import java.util.Scanner;
public class ex004 {
	public static void main(String[] args) {
		Scanner num = new Scanner(System.in);
		int num1, num2, num3, max, min;
		System.out.print("請輸入3個數字(用空白鍵區隔) : ");
		num1 = num.nextInt();
		num2 = num.nextInt();
		num3 = num.nextInt();
		
		max = min = num1;
		if (num1 > max){
			max = num1;
		}else if (num1 < min){ 
			min = num1;
		}
		if (num2 > max){
			max = num2;
		}else if (num2 < min){
			min = num2;
		}
		if (num3 > max){
			max = num3;
		}else if (num3 < min){
			min = num3;
		}
		System.out.printf("最大的數字是 : %d%n最小的數字是 : %d%n",max ,min);
		num.close();
	}
}

# 5個數字同時取最大與最小
```
import java.util.Scanner;
public class ex006 {
public static void main(String[] args) {
		Scanner n = new Scanner(System.in);
		int max, min;
		System.out.print("請輸入5個數字(用空白鍵區隔): ");
		int n1 = n.nextInt();
		int n2 = n.nextInt();
		int n3 = n.nextInt();
		int n4 = n.nextInt();
		int n5 = n.nextInt();
		
		
		int arr[] = {n1, n2, n3, n4, n5};
		max =  min = arr[0];
		for (int i = 0; i <= 4; i++){
			if (arr[i] >= max){
				max = arr[i];
			}else if (arr[i] <= min){
				min = arr[i];
			}
		}
		System.out.printf("最大的數字是 : %d%n最小的數字是 : %d%n",max ,min);
		n.close();
	}
}
